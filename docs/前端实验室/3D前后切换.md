    ; (function () {
        var prev = document.querySelector('.wireless-lab-bigbang-3dqianhouqiehuan-1 .prev');
        var next = document.querySelector('.wireless-lab-bigbang-3dqianhouqiehuan-1 .next');

        var items = document.querySelectorAll('.wireless-lab-bigbang-3dqianhouqiehuan-1 li');

        var allClass = [];

        var isReady = true;

        for (var i = 0, len = items.length; i < len; i++) {
            allClass.push(items[i].className);
        }

        var updateClass = function () {
            for (var i = 0, len = items.length; i < len; i++) {
                items[i].className = allClass[i];
            }

            document.querySelector('.wireless-lab-bigbang-3dqianhouqiehuan-1 .cur').addEventListener('transitionend', function () {
                isReady = true;
            }, false);
        };

        prev.onclick = function () {
            if (!isReady) {
                return;
            }

            isReady = false;

            allClass.unshift(allClass.pop());

            console.log(allClass);
            updateClass();
        };

        next.onclick = function () {
            if (!isReady) {
                return;
            }

            isReady = false;

            allClass.push(allClass.shift());

            updateClass();
        };

    }());
</script>
